Ping Hostsvalidated REST endpoint with persistence
@app.post("/hosts", response_model=Host)
def add_host(host: Host, db: Session = Depends(get_db)):
    duplicate = db.query(HostDB).filter(
        HostDB.ip == str(host.ip)
    ).first()
    if duplicate:
        raise HTTPException(
            status_code=400, detail="IP already exists"
        )
    db_host = HostDB(id=host.id, ip=str(host.ip))
    db.add(db_host)
    db.commit()
    db.refresh(db_host)
    return db_host
